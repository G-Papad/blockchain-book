      \Let{(y, \pi)}{\vrfeval(\sk, \slot)}

        \Let{B}{H(\chain[-1].B) \concat x \concat \slot \concat \pk \concat y \concat \pi}
        \Let{\sigma}{\Sig(\sk, B)}
        \Let{\chain}{\chain \concat (B, \sigma)}
      \EndIf
      \State $\broadcast(\chain)$
      \Let{\slot}{\slot + 1}
    \EndFunction
