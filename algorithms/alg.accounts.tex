\begin{algorithm}[H]
  \caption{\label{alg.accounts}
  The accounts transition function.}
  \begin{algorithmic}[1]
    \Function{$\delta_\textsf{account}$}{$\st, \tx$}
      \CommentLine{Check sufficient balance}
      \If{$\tx.\textsf{from} \not\in \st \lor \st[\tx.\textsf{from}].\textsf{balance} < \tx.\textsf{value}$}
        \State\Return$\bot$
      \EndIf
      \If{$\tx.\textsf{nonce} \neq \st[\tx.\textsf{from}].\textsf{nonce} + 1$}\Comment{Check nonce is unique}
        \State\Return$\bot$
      \EndIf
      \Let{m}{\{
        \textsf{from}\colon \tx.\textsf{from},
        \textsf{to}\colon \tx.\textsf{to},
        \textsf{value}\colon \tx.\textsf{value},
        \textsf{nonce}\colon \tx.\textsf{nonce},
        \sigma\colon \bot
      \}}
      \If{$\lnot\Ver(\tx.\textsf{from}, \tx.\sigma, m)$}
        \Comment{Check signature}
        \State\Return$\bot$
      \EndIf
      \Let{\st[\tx.\textsf{from}].\textsf{balance}}{\st[\tx.\textsf{from}].\textsf{balance} - \tx.\textsf{value}}
      \Let{\st[\tx.\textsf{from}].\textsf{nonce}}{\st[\tx.\textsf{from}].\textsf{nonce} + 1}
      \If{$\tx.\textsf{to} \not\in \st$}\Comment{Check account exists}
        \Let{\st[\tx.\textsf{to}]}{\{\textsf{balance}\colon 0, \textsf{nonce}\colon 0\}}
      \EndIf
      \Let{\st[\tx.\textsf{to}].\textsf{balance}}{\st[\tx.\textsf{to}].\textsf{balance} + \tx.\textsf{value}}
      \State\Return$\st$
    \EndFunction
  \end{algorithmic}
\end{algorithm}
