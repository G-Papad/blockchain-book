\begin{algorithm}[H]
  \caption{\label{alg.utxo}
  The UTXO transition function.}
  \begin{algorithmic}[1]
    \Function{$\delta_\textsf{UTXO}$}{$\st, \tx$}
      \Let{m}{\{\inputs\colon[\,], \outputs\colon \tx.\outputs\}}
      \Comment{Create unsigned transaction}
      \For{$\textsf{in} \in \tx.\inputs$}
        \Let{(\outpoint, \_)}{\textsf{in}}
        \Let{m.\inputs}{m.\inputs \concat (\outpoint, \bot)}
      \EndFor
      \Let{\st'}{\st}
      \For{$\textsf{in} \in \tx.\inputs$}
        \Let{(\textsf{outpoint}, \sigma)}{\textsf{in}}
        \If{$\outpoint \not\in \st'$}
          \Comment{Check that input is unspent}
          \State\Return$\bot$
        \EndIf
        \If{$\lnot\Ver(\st'[\outpoint].\textsf{pk}, \sigma, m)$}
          \Comment{Check that signature is valid}
          \State\Return$\bot$
        \EndIf
        \Let{\st'[\outpoint]}{\bot}\Comment{Consume UTXO}
      \EndFor
      \CommentLine{Check the Conservation Law}
      \If{$\sum_{(\outpoint, \_) \in \tx.\inputs} \st[\outpoint].\amount < \sum_{\textsf{out} \in \tx.\outputs} \textsf{out}.\amount$}
        \State\Return$\bot$
      \EndIf
      \Let{\idx}{0}
      \Let{\txid}{H(\tx)}
      \For{$\textsf{out} \in \tx.\outputs$}
        \Let{\outpoint}{(\txid, \idx)}\Comment{Create new UTXO}
        \Let{\st'[\outpoint]}{\textsf{out}}
        \Let{\idx}{\idx + 1}
      \EndFor
      \State\Return$\st'$
    \EndFunction
  \end{algorithmic}
\end{algorithm}
